// ... (Keep your Audio constants and gameData initialization at the top) let activeQuestions = []; // This will hold our shuffled pool let isProcessing = false; function renderUI() { isProcessing = false; // 1. Safety Check: Is the data file loaded? if (typeof quizDatabase === 'undefined' || quizDatabase.length === 0) { document.getElementById('quiz-container').innerHTML = `
QUESTIONS.JS NOT LOADED
Make sure your file is named 'questions.js' and is in the same folder.
`; return; } // 2. Find Current Level (or fall back to the last available one) let levelData = quizDatabase.find(l => l.level === gameData.lvl); if (!levelData) { levelData = quizDatabase[quizDatabase.length - 1]; // Loop at the last level } // 3. SHUFFLE LOGIC: If pool is empty or we finished the current list, REFILL IT if (activeQuestions.length === 0 || gameData.currentQuestionIndex >= activeQuestions.length) { // We clone and shuffle the questions from the current levelData activeQuestions = [...levelData.questions].sort(() => Math.random() - 0.5); gameData.currentQuestionIndex = 0; console.log("Shuffled new pool of " + activeQuestions.length + " questions."); } // 4. Update Visuals document.getElementById('bg-overlay').style.backgroundImage = `url('${levelData.bgImage}')`; document.getElementById('world-title').innerText = levelData.name; document.getElementById('lvl-badge').innerText = gameData.lvl; document.getElementById('gold-val').innerText = gameData.gold; document.getElementById('xp-bar').style.width = (gameData.currentQuestionIndex / activeQuestions.length * 100) + "%"; renderCurrentQuestionText(); startTimer(); } function handleAnswer(selected, correct, btn) { if (isProcessing) return; isProcessing = true; clearInterval(timerInterval); if (selected === correct) { playSfx(soundCorrect); btn.style.background = "#dcfce7"; btn.style.borderColor = "#22c55e"; gameData.gold += 20; // Move to next question index gameData.currentQuestionIndex++; // Check if we just finished the last question in this level's pool if (gameData.currentQuestionIndex >= activeQuestions.length) { // Check if there is a next level defined in questions.js const nextLevelExists = quizDatabase.some(l => l.level === gameData.lvl + 1); if (nextLevelExists) { gameData.lvl++; gameData.currentQuestionIndex = 0; activeQuestions = []; // Force a fresh shuffle for the new level save(); setTimeout(showLevelUp, 600); } else { // No more levels? Just reshuffle the current one and keep going activeQuestions = []; gameData.currentQuestionIndex = 0; save(); setTimeout(() => { timeLeft = 10; renderUI(); }, 800); } } else { save(); setTimeout(() => { timeLeft = 10; renderUI(); }, 800); } } else { // WRONG ANSWER LOGIC playSfx(soundWrong); btn.style.background = "#fee2e2"; btn.style.borderColor = "#ef4444"; gameData.gold = Math.max(0, gameData.gold - 10); save(); // Let them try the same question again setTimeout(() => { isProcessing = false; timeLeft = 10; renderUI(); }, 800); } }
